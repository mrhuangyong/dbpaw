pub struct Db2Driver {
    config: Db2Config,

#[derive(Clone)]
struct Db2Config {
    host: String,
    port: u16,
    database: String,
    username: String,
    password: String,
}

fn build_connection_string(cfg: &Db2Config) -> String {
    format!(
        "DRIVER={{IBM DB2 ODBC DRIVER}};DATABASE={};HOSTNAME={};PORT={};PROTOCOL=TCPIP;UID={};PWD={};",
        cfg.database, cfg.host, cfg.port, cfg.username, cfg.password
    )
}

fn quote_ident(ident: &str) -> String {
    format!("\"{}\"", ident.replace('"', "\"\""))
}

fn escape_literal(value: &str) -> String {
    value.replace('\'', "''")
}

fn odbc_value_to_json(row: &odbc_api::CursorImpl<&Connection>, idx: usize) -> serde_json::Value {
    use odbc_api::buffers::TextRowSet;
    match row.get_text(idx) {
        None => serde_json::Value::Null,
        Some(bytes) => {
            let s = String::from_utf8_lossy(bytes).to_string();
            if s.is_empty() {
                return serde_json::Value::Null;
            }
            if let Ok(n) = s.parse::<i64>() {
                return serde_json::Value::Number(n.into());
            }
            if let Ok(f) = s.parse::<f64>() {
                if let Some(n) = serde_json::Number::from_f64(f) {
                    return serde_json::Value::Number(n);
                }
            }
            serde_json::Value::String(s)
        }
    }
}

impl Db2Driver {
    pub async fn connect(form: &ConnectionForm) -> Result<Self, String> {
        let mut effective_form = form.clone();
        let mut ssh_tunnel = None;

        if let Some(true) = form.ssh_enabled {
            let tunnel = crate::ssh::start_ssh_tunnel(form)?;
            effective_form.host = Some("127.0.0.1".to_string());
            effective_form.port = Some(tunnel.local_port as i64);
            ssh_tunnel = Some(tunnel);
        }

        let host = effective_form
            .host
            .clone()
            .map(|v| v.trim().to_string())
            .filter(|v| !v.is_empty())
            .ok_or("[VALIDATION_ERROR] host cannot be empty")?;
        let port = effective_form.port.unwrap_or(50000);
        if !(1..=65535).contains(&port) {
            return Err("[VALIDATION_ERROR] port out of range".to_string());
        }
        let database = effective_form
            .database
            .clone()
            .map(|v| v.trim().to_string())
            .filter(|v| !v.is_empty())
            .ok_or("[VALIDATION_ERROR] database cannot be empty")?;
        let username = effective_form
            .username
            .clone()
            .map(|v| v.trim().to_string())
            .filter(|v| !v.is_empty())
            .ok_or("[VALIDATION_ERROR] username cannot be empty")?;
        let password = effective_form.password.clone().unwrap_or_default();

        let config = Db2Config {
            host,
            port: port as u16,
            database,
            username,
            password,
        };
        let driver = Self {
            config,
            _ssh_tunnel: ssh_tunnel,
        };
        driver.test_connection().await?;
        Ok(driver)
    }

    async fn run_blocking<F, T>(&self, f: F) -> Result<T, String>
    where
        F: FnOnce(odbc_api::Connection) -> Result<T, String> + Send + 'static,
        T: Send + 'static,
    {
        let cfg = self.config.clone();
        tokio::task::spawn_blocking(move || {
            let conn_string = build_connection_string(&cfg);
            let mut env = Environment::new()
                .map_err(|e| format!("[CONN_FAILED] Failed to create ODBC environment: {e}"))?;
            let conn = env
                .connect_with_connection_string(&conn_string)
                .map_err(|e| conn_failed_error(&e))?;
            f(conn)
        })
        .await
        .map_err(|e| format!("[DB2_ERROR] {e}"))?
    }
}

        })

    }
}

    async fn list_tables(&self, schema: Option<String>) -> Result<Vec<TableInfo>, String> {
        let schema_upper = schema
            .map(|s| s.trim().to_uppercase())
            .filter(|s| !s.is_empty());
        self.run_blocking(move |conn| {
            let sql = if let Some(ref s) = schema_upper {
                format!(
                    "SELECT TABSCHEMA, TABNAME, CASE WHEN TYPE = 'V' THEN 'view' ELSE 'table' END \
                     FROM SYSCAT.TABLES \
                     WHERE TABSCHEMA = '{}' AND TYPE IN ('T', 'V') \
                     ORDER BY TABSCHEMA, TABNAME",
                    escape_literal(s)
                )
            } else {
                "SELECT TABSCHEMA, TABNAME, CASE WHEN TYPE = 'V' THEN 'view' ELSE 'table' END \
                 FROM SYSCAT.TABLES \
                 WHERE TABSCHEMA NOT LIKE 'SYS%' AND TYPE IN ('T', 'V') \
                 ORDER BY TABSCHEMA, TABNAME"
                    .to_string()
            };
            let mut stmt = conn
                .prepare(&sql)
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?;
            let mut cursor = stmt
                .execute(&[])
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?
                .ok_or("[QUERY_ERROR] No result")?;
            let mut result = Vec::new();
            while let Some(row) = cursor.next().map_err(|e| format!("[QUERY_ERROR] {e}"))? {
                let schema_name = row.get_text(0).map(|b| String::from_utf8_lossy(b).to_string());
                let table_name = row.get_text(1).map(|b| String::from_utf8_lossy(b).to_string());
                let table_type = row.get_text(2).map(|b| String::from_utf8_lossy(b).to_string());
                if let (Some(s), Some(t), Some(ty)) = (schema_name, table_name, table_type) {
                    result.push(TableInfo {
                        schema: s,
                        name: t,
                        r#type: ty,
                    });
                }
            }
            Ok(result)
        })
        .await
    }

    async fn get_table_structure(
        &self,
        schema: String,
        table: String,
    ) -> Result<TableStructure, String> {
        self.run_blocking(move |conn| {
            let pk_sql = format!(
                "SELECT COLNAME \
                 FROM SYSCAT.KEYCOLUSE \
                 WHERE TABSCHEMA = '{}' AND TABNAME = '{}' \
                 ORDER BY COLSEQ",
                escape_literal(&schema.to_uppercase()),
                escape_literal(&table.to_uppercase()),
            );
            let mut pk_set = std::collections::HashSet::<String>::new();
            let mut stmt = conn
                .prepare(&pk_sql)
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?;
            if let Some(mut cursor) = stmt
                .execute(&[])
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?
            {
                while let Some(row) = cursor.next().map_err(|e| format!("[QUERY_ERROR] {e}"))? {
                    if let Some(bytes) = row.get_text(0) {
                        let col = String::from_utf8_lossy(bytes).to_string();
                        if !col.is_empty() {
                            pk_set.insert(col);
                        }
                    }
                }
            }

            let col_sql = format!(
                "SELECT \
                    COLNAME, \
                    TYPENAME || \
                    CASE \
                        WHEN TYPENAME IN ('VARCHAR','NVARCHAR','CHAR','NCHAR','VARGRAPHIC') \
                            THEN '(' || LENGTH || ')' \
                        WHEN TYPENAME = 'DECIMAL' \
                            THEN '(' || LENGTH || ',' || SCALE || ')' \
                        ELSE '' \
                    END AS FULL_TYPE, \
                    NULLS, \
                    DEFAULT, \
                    REMARKS \
                 FROM SYSCAT.COLUMNS \
                 WHERE TABSCHEMA = '{}' AND TABNAME = '{}' \
                 ORDER BY COLNO",
                escape_literal(&schema.to_uppercase()),
                escape_literal(&table.to_uppercase()),
            );
            let mut columns = Vec::new();
            let mut stmt = conn
                .prepare(&col_sql)
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?;
            if let Some(mut cursor) = stmt
                .execute(&[])
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?
            {
                while let Some(row) = cursor.next().map_err(|e| format!("[QUERY_ERROR] {e}"))? {
                    let name = row.get_text(0).map(|b| String::from_utf8_lossy(b).to_string());
                    let col_type = row.get_text(1).map(|b| String::from_utf8_lossy(b).to_string());
                    let nullable = row.get_text(2).map(|b| String::from_utf8_lossy(b).to_string());
                    let default_val = row.get_text(3).map(|b| String::from_utf8_lossy(b).to_string());
                    let remark = row.get_text(4).map(|b| String::from_utf8_lossy(b).to_string());
                    if let (Some(name), Some(col_type)) = (name, col_type) {
                        let is_nullable = nullable.as_deref() != Some("N");
                        let default_value = default_val
                            .map(|d| d.trim().to_string())
                            .filter(|d| !d.is_empty());
                        let comment = remark.filter(|r| !r.is_empty());
                        let primary_key = pk_set.contains(&name);
                        columns.push(ColumnInfo {
                            name,
                            r#type: col_type,
                            nullable: is_nullable,
                            default_value,
                            primary_key,
                            comment,
                            default_constraint_name: None,
                        });
                    }
                }
            }
            Ok(TableStructure { columns })
        })
        .await
    }

    async fn get_table_metadata(
        &self,
        schema: String,
        table: String,
    ) -> Result<TableMetadata, String> {
        let columns = self
            .get_table_structure(schema.clone(), table.clone())
            .await?
            .columns;

        let (indexes, foreign_keys) = self
            .run_blocking(move |conn| {
                let idx_sql = format!(
                    "SELECT i.INDNAME, \
                            CASE WHEN i.UNIQUERULE = 'U' OR i.UNIQUERULE = 'P' THEN 1 ELSE 0 END, \
                            i.INDEXTYPE, \
                            ic.COLNAME, \
                            ic.COLSEQ \
                     FROM SYSCAT.INDEXES i \
                     JOIN SYSCAT.INDEXCOLUSE ic \
                       ON ic.INDNAME = i.INDNAME \
                      AND IC.INDSCHEMA = i.INDSCHEMA \
                     WHERE i.TABSCHEMA = '{}' \
                       AND i.TABNAME = '{}' \
                     ORDER BY i.INDNAME, ic.COLSEQ",
                    escape_literal(&schema.to_uppercase()),
                    escape_literal(&table.to_uppercase()),
                );
                let mut stmt = conn
                    .prepare(&idx_sql)
                    .map_err(|e| format!("[QUERY_ERROR] {e}"))?;
                let mut idx_map: HashMap<String, (bool, Option<String>, Vec<(i64, String)>)> =
                    HashMap::new();
                if let Some(mut cursor) = stmt
                    .execute(&[])
                    .map_err(|e| format!("[QUERY_ERROR] {e}"))?
                {
                    while let Some(row) =
                        cursor.next().map_err(|e| format!("[QUERY_ERROR] {e}"))?
                    {
                        let idx_name =
                            row.get_text(0).map(|b| String::from_utf8_lossy(b).to_string());
                        let is_unique: Option<i64> = row
                            .get_text(1)
                            .and_then(|b| String::from_utf8_lossy(b).parse().ok());
                        let idx_type =
                            row.get_text(2).map(|b| String::from_utf8_lossy(b).to_string());
                        let col_name =
                            row.get_text(3).map(|b| String::from_utf8_lossy(b).to_string());
                        let position: Option<i64> = row
                            .get_text(4)
                            .and_then(|b| String::from_utf8_lossy(b).parse().ok());
                        if let (Some(name), Some(col_name)) = (idx_name, col_name) {
                            let unique = is_unique.unwrap_or(0) == 1;
                            let pos = position.unwrap_or(0);
                            let entry = idx_map.entry(name).or_insert((unique, None, Vec::new()));
                            entry.0 = unique;
                            if entry.1.is_none() {
                                entry.1 = idx_type;
                            }
                            entry.2.push((pos, col_name));
                        }
                    }
                }
                let mut indexes: Vec<IndexInfo> = idx_map
                    .into_iter()
                    .map(|(name, (unique, index_type, mut cols))| {
                        cols.sort_by_key(|(pos, _)| *pos);
                        IndexInfo {
                            name,
                            unique,
                            index_type,
                            columns: cols.into_iter().map(|(_, c)| c).collect(),
                        }
                    })
                    .collect();
                indexes.sort_by(|a, b| a.name.cmp(&b.name));

                let fk_sql = format!(
                    "SELECT fk.CONSTNAME, \
                            fk.COLNAME, \
                            reftab.TABSCHEMA, \
                            reftab.TABNAME, \
                            refcol.COLNAME, \
                            fk.DELETERULE \
                     FROM SYSCAT.REFERENCES fk \
                     JOIN SYSCAT.TABLES reftab \
                       ON reftab.TABNAME = fk.REFTABNAME \
                      AND reftab.TABSCHEMA = fk.REFTABSCHEMA \
                     JOIN SYSCAT.KEYCOLUSE refcol \
                       ON refcol.CONSTNAME = fk.REFKEYNAME \
                      AND refcol.TABSCHEMA = fk.REFTABSCHEMA \
                     WHERE fk.TABSCHEMA = '{}' \
                       AND fk.TABNAME = '{}' \
                     ORDER BY fk.CONSTNAME, refcol.COLSEQ",
                    escape_literal(&schema.to_uppercase()),
                    escape_literal(&table.to_uppercase()),
                );
                let mut foreign_keys = Vec::new();
                let mut stmt = conn
                    .prepare(&fk_sql)
                    .map_err(|e| format!("[QUERY_ERROR] {e}"))?;
                if let Some(mut cursor) = stmt
                    .execute(&[])
                    .map_err(|e| format!("[QUERY_ERROR] {e}"))?
                {
                    while let Some(row) =
                        cursor.next().map_err(|e| format!("[QUERY_ERROR] {e}"))?
                    {
                        let fk_name =
                            row.get_text(0).map(|b| String::from_utf8_lossy(b).to_string());
                        let col_name =
                            row.get_text(1).map(|b| String::from_utf8_lossy(b).to_string());
                        let ref_schema =
                            row.get_text(2).map(|b| String::from_utf8_lossy(b).to_string());
                        let ref_table =
                            row.get_text(3).map(|b| String::from_utf8_lossy(b).to_string());
                        let ref_col =
                            row.get_text(4).map(|b| String::from_utf8_lossy(b).to_string());
                        let delete_rule =
                            row.get_text(5).map(|b| String::from_utf8_lossy(b).to_string());
                        if let (Some(fk_name), Some(col_name), Some(ref_table), Some(ref_col)) =
                            (fk_name, col_name, ref_table, ref_col)
                        {
                            foreign_keys.push(ForeignKeyInfo {
                                name: fk_name,
                                column: col_name,
                                referenced_schema: ref_schema,
                                referenced_table: ref_table,
                                referenced_column: ref_col,
                                on_update: None,
                                on_delete: delete_rule,
                            });
                        }
                    }
                }
                Ok((indexes, foreign_keys))
            })
            .await?;

        Ok(TableMetadata {
            columns,
            indexes,
            foreign_keys,
            clickhouse_extra: None,
            cassandra_extra: None,
            special_type_summaries: vec![],
        })
    }

    async fn get_table_ddl(&self, schema: String, table: String) -> Result<String, String> {
        self.run_blocking(move |conn| {
            let col_sql = format!(
                "SELECT COLNAME, TYPENAME, LENGTH, SCALE, NULLS, DEFAULT, IDENTITY, GENERATED \
                 FROM SYSCAT.COLUMNS \
                 WHERE TABSCHEMA = '{}' AND TABNAME = '{}' \
                 ORDER BY COLNO",
                escape_literal(&schema.to_uppercase()),
                escape_literal(&table.to_uppercase()),
            );
            let mut stmt = conn
                .prepare(&col_sql)
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?;
            let mut columns = Vec::new();
            if let Some(mut cursor) = stmt
                .execute(&[])
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?
            {
                while let Some(row) = cursor.next().map_err(|e| format!("[QUERY_ERROR] {e}"))? {
                    let col_name = row.get_text(0).map(|b| String::from_utf8_lossy(b).to_string());
                    let type_name = row.get_text(1).map(|b| String::from_utf8_lossy(b).to_string());
                    let length: Option<i64> = row.get_text(2).and_then(|b| String::from_utf8_lossy(b).parse().ok());
                    let scale: Option<i64> = row.get_text(3).and_then(|b| String::from_utf8_lossy(b).parse().ok());
                    let nulls = row.get_text(4).map(|b| String::from_utf8_lossy(b).to_string());
                    let default = row.get_text(5).map(|b| String::from_utf8_lossy(b).to_string());
                    let identity = row.get_text(6).map(|b| String::from_utf8_lossy(b).to_string());
                    let generated = row.get_text(7).map(|b| String::from_utf8_lossy(b).to_string());
                    if let (Some(name), Some(ty)) = (col_name, type_name) {
                        columns.push((name, ty, length, scale, nulls, default, identity, generated));
                    }
                }
            }

            let pk_sql = format!(
                "SELECT COLNAME FROM SYSCAT.KEYCOLUSE \
                 WHERE TABSCHEMA = '{}' AND TABNAME = '{}' ORDER BY COLSEQ",
                escape_literal(&schema.to_uppercase()),
                escape_literal(&table.to_uppercase()),
            );
            let mut pk_cols = Vec::new();
            let mut stmt = conn
                .prepare(&pk_sql)
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?;
            if let Some(mut cursor) = stmt
                .execute(&[])
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?
            {
                while let Some(row) = cursor.next().map_err(|e| format!("[QUERY_ERROR] {e}"))? {
                    if let Some(bytes) = row.get_text(0) {
                        pk_cols.push(String::from_utf8_lossy(bytes).to_string());
                    }
                }
            }

            let mut ddl = format!(
                "CREATE TABLE {}.{} (\n",
                quote_ident(&schema.to_uppercase()),
                quote_ident(&table.to_uppercase())
            );
            let mut col_defs = Vec::new();
            for (name, ty, length, scale, nulls, default, identity, generated) in &columns {
                let mut def = format!("  {} {}", quote_ident(name), ty);
                if let Some(len) = length {
                    if matches!(ty.as_str(), "VARCHAR" | "NVARCHAR" | "CHAR" | "NCHAR" | "VARGRAPHIC") {
                        def.push_str(&format!("({})", len));
                    } else if ty == "DECIMAL" {
                        let s = scale.unwrap_or(0);
                        def.push_str(&format!("({},{})", len, s));
                    }
                }
                if identity.as_deref() == Some("Y") {
                    def.push_str(" GENERATED ALWAYS AS IDENTITY");
                }
                if nulls.as_deref() == Some("N") {
                    def.push_str(" NOT NULL");
                }
                if let Some(d) = default {
                    let d = d.trim();
                    if !d.is_empty() && d != "NULL" {
                        def.push_str(&format!(" DEFAULT {}", d));
                    }
                }
                col_defs.push(def);
            }
            if !pk_cols.is_empty() {
                let pk_list: Vec<String> = pk_cols.iter().map(|c| quote_ident(c)).collect();
                col_defs.push(format!("  PRIMARY KEY ({})", pk_list.join(", ")));
            }
            ddl.push_str(&col_defs.join(",\n"));
            ddl.push_str("\n);");
            Ok(ddl)
        })
        .await
    }

    async fn get_table_data(
        &self,
        schema: String,
        table: String,
        page: i64,
        limit: i64,
        sort_column: Option<String>,
        sort_direction: Option<String>,
        filter: Option<String>,
        order_by: Option<String>,
    ) -> Result<TableDataResponse, String> {
        let start = std::time::Instant::now();
        let safe_page = if page < 1 { 1 } else { page };
        let safe_limit = if limit < 1 { 100 } else { limit };
        let offset = (safe_page - 1) * safe_limit;

        let filter = filter.map(|f| super::normalize_quotes(&f));
        let order_by = order_by.map(|f| super::normalize_quotes(&f));

        let table_ref = format!(
            "{}.{}",
            quote_ident(&schema.to_uppercase()),
            quote_ident(&table.to_uppercase())
        );

        let where_clause = match &filter {
            Some(f) if !f.trim().is_empty() => format!(" WHERE {}", f.trim()),
            _ => String::new(),
        };

        let order_clause = if let Some(ref raw) = order_by {
            if raw.trim().is_empty() {
                String::new()
            } else {
                format!(" ORDER BY {}", raw.trim())
            }
        } else if let Some(ref col) = sort_column {
            let dir = if matches!(sort_direction.as_deref(), Some("desc")) {
                "DESC"
            } else {
                "ASC"
            };
            format!(" ORDER BY {} {}", quote_ident(col), dir)
        } else {
            String::new()
        };

        self.run_blocking(move |conn| {
            let count_sql = format!("SELECT COUNT(*) FROM {}{}", table_ref, where_clause);
            let mut stmt = conn
                .prepare(&count_sql)
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?;
            let mut total: i64 = 0;
            if let Some(mut cursor) = stmt
                .execute(&[])
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?
            {
                if let Some(row) = cursor.next().map_err(|e| format!("[QUERY_ERROR] {e}"))? {
                    total = row
                        .get_text(0)
                        .and_then(|b| String::from_utf8_lossy(b).trim().parse().ok())
                        .unwrap_or(0);
                }
            }

            let data_sql = format!(
                "SELECT * FROM {}{}{} LIMIT {} OFFSET {}",
                table_ref, where_clause, order_clause, safe_limit, offset
            );
            let mut stmt = conn
                .prepare(&data_sql)
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?;
            let mut cursor = stmt
                .execute(&[])
                .map_err(|e| format!("[QUERY_ERROR] {e}"))?
                .ok_or("[QUERY_ERROR] No result")?;

            let col_count = cursor.num_result_cols().map_err(|e| format!("[QUERY_ERROR] {e}"))?;
            let mut col_names = Vec::new();
            for i in 1..=col_count {
                let name = cursor
                    .col_name(i as u16)
                    .map_err(|e| format!("[QUERY_ERROR] {e}"))?
                    .unwrap_or_else(|| format!("col_{}", i))
                    .to_string();
                col_names.push(name);
            }

            let mut data = Vec::new();
            while let Some(row) = cursor.next().map_err(|e| format!("[QUERY_ERROR] {e}"))? {
                let mut map = serde_json::Map::new();
                for (i, name) in col_names.iter().enumerate() {
                    map.insert(name.clone(), odbc_value_to_json_from_row(&row, i));
                }
                data.push(serde_json::Value::Object(map));
            }

            Ok(TableDataResponse {
                data,
                total,
                page: safe_page,
                limit: safe_limit,
                execution_time_ms: start.elapsed().as_millis() as i64,
            })
        })
        .await
    }

use dbpaw_lib::db::drivers::DatabaseDriver;
use dbpaw_lib::models::ConnectionForm;

pub async fn connect_with_retry<F, Fut>(connect_fn: F) -> Box<dyn DatabaseDriver>
where
    F: Fn() -> Fut,
    Fut: std::future::Future<Output = Result<Box<dyn DatabaseDriver>, String>>,
{
    let mut last_err = String::new();
    for attempt in 1..=30 {
        match connect_fn().await {
            Ok(driver) => return driver,
            Err(e) => {
                last_err = e;
                if attempt < 30 {
                    tokio::time::sleep(std::time::Duration::from_secs(2)).await;
                }
            }
        }
    }
    panic!("Failed to connect to Db2 after 30 attempts: {}", last_err);
}

pub fn db2_connection_form() -> ConnectionForm {
    ConnectionForm {
        driver: "db2".to_string(),
        host: Some("127.0.0.1".to_string()),
        port: Some(50000),
        database: Some("testdb".to_string()),
        username: Some("db2inst1".to_string()),
        password: Some("testpass".to_string()),
        ..Default::default()
    }
}
